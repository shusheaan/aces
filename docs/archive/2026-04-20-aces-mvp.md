# ACES (Air Combat Engagement Simulation) MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a 1v1 air combat simulation where two RL agents learn to pursue and evade each other in a bounded 3D arena with obstacles, with real-time 3D visualization and online learning.

**Architecture:** Rust core (PyO3/maturin) handles performance-critical state management, collision detection, and reward computation. Python handles flight dynamics (simplified 6-DOF physics, pluggable for JSBSim), Gymnasium environment wrapper, PPO self-play training via stable-baselines3, and 3D visualization via PyVista. Training and visualization run concurrently — the user watches agents improve in real time.

**Tech Stack:** Rust 1.75+ / PyO3 0.22 / maturin, Python 3.11+, Gymnasium 0.29+, stable-baselines3 2.x (PPO), numpy, PyVista 0.43+, toml

// src/state.rs
use pyo3::prelude::*;

/// State of a single aircraft in the simulation.
#[pyclass]
#[derive(Clone, Debug)]
pub struct AircraftState {
    #[pyo3(get, set)]
    pub x: f64,
    #[pyo3(get, set)]
    pub y: f64,
    #[pyo3(get, set)]
    pub z: f64,
    #[pyo3(get, set)]
    pub vx: f64,
    #[pyo3(get, set)]
    pub vy: f64,
    #[pyo3(get, set)]
    pub vz: f64,
    #[pyo3(get, set)]
    pub roll: f64,
    #[pyo3(get, set)]
    pub pitch: f64,
    #[pyo3(get, set)]
    pub yaw: f64,
}

#[pymethods]
impl AircraftState {
    #[new]
    pub fn new(
        x: f64, y: f64, z: f64,
        vx: f64, vy: f64, vz: f64,
        roll: f64, pitch: f64, yaw: f64,
    ) -> Self {
        Self { x, y, z, vx, vy, vz, roll, pitch, yaw }
    }

    pub fn position(&self) -> [f64; 3] {
        [self.x, self.y, self.z]
    }

    pub fn velocity(&self) -> [f64; 3] {
        [self.vx, self.vy, self.vz]
    }

    pub fn orientation(&self) -> [f64; 3] {
        [self.roll, self.pitch, self.yaw]
    }

    pub fn speed(&self) -> f64 {
        (self.vx * self.vx + self.vy * self.vy + self.vz * self.vz).sqrt()
    }

    /// Forward unit vector in world frame (Z-up, yaw=0 → +X).
    pub fn forward_vector(&self) -> [f64; 3] {
        let (sp, cp) = self.pitch.sin_cos();
        let (sy, cy) = self.yaw.sin_cos();
        [cy * cp, sy * cp, sp]
    }

    pub fn to_array(&self) -> [f64; 9] {
        [self.x, self.y, self.z, self.vx, self.vy, self.vz, self.roll, self.pitch, self.yaw]
    }
}

impl AircraftState {
    pub fn distance_to(&self, other: &AircraftState) -> f64 {
        let dx = other.x - self.x;
        let dy = other.y - self.y;
        let dz = other.z - self.z;
        (dx * dx + dy * dy + dz * dz).sqrt()
    }

    pub fn angle_to(&self, other: &AircraftState) -> f64 {
        let dx = other.x - self.x;
        let dy = other.y - self.y;
        let dz = other.z - self.z;
        let dist = (dx * dx + dy * dy + dz * dz).sqrt();
        if dist < 1e-9 {
            return 0.0;
        }
        let to_other = [dx / dist, dy / dist, dz / dist];
        let fwd = self.forward_vector();
        let dot = fwd[0] * to_other[0] + fwd[1] * to_other[1] + fwd[2] * to_other[2];
        dot.clamp(-1.0, 1.0).acos()
    }
}
```

// src/arena.rs
use pyo3::prelude::*;
use crate::state::AircraftState;

#[pyclass]
#[derive(Clone, Debug)]
pub struct Obstacle {
    #[pyo3(get)]
    pub x: f64,
    #[pyo3(get)]
    pub y: f64,
    #[pyo3(get)]
    pub z: f64,
    #[pyo3(get)]
    pub radius: f64,
}

#[pymethods]
impl Obstacle {
    #[new]
    pub fn new(x: f64, y: f64, z: f64, radius: f64) -> Self {
        Self { x, y, z, radius }
    }

    pub fn position(&self) -> [f64; 3] {
        [self.x, self.y, self.z]
    }
}

#[pyclass]
#[derive(Clone, Debug)]
pub struct Arena {
    #[pyo3(get)]
    pub x_max: f64,
    #[pyo3(get)]
    pub y_max: f64,
    #[pyo3(get)]
    pub z_max: f64,
    obstacles: Vec<Obstacle>,
}

#[pymethods]
impl Arena {
    #[new]
    pub fn new(x_max: f64, y_max: f64, z_max: f64) -> Self {
        Self { x_max, y_max, z_max, obstacles: Vec::new() }
    }

    pub fn add_obstacle(&mut self, obs: Obstacle) {
        self.obstacles.push(obs);
    }

    pub fn get_obstacles(&self) -> Vec<Obstacle> {
        self.obstacles.clone()
    }

    pub fn num_obstacles(&self) -> usize {
        self.obstacles.len()
    }

    pub fn is_in_bounds(&self, x: f64, y: f64, z: f64) -> bool {
        x >= 0.0 && x <= self.x_max
            && y >= 0.0 && y <= self.y_max
            && z >= 0.0 && z <= self.z_max
    }

    pub fn check_obstacle_collision(&self, x: f64, y: f64, z: f64) -> bool {
        for obs in &self.obstacles {
            let dx = x - obs.x;
            let dy = y - obs.y;
            let dz = z - obs.z;
            if dx * dx + dy * dy + dz * dz < obs.radius * obs.radius {
                return true;
            }
        }
        false
    }

    /// Returns (out_of_bounds, hit_obstacle) for an aircraft state.
    pub fn check_state(&self, state: &AircraftState) -> (bool, bool) {
        let oob = !self.is_in_bounds(state.x, state.y, state.z);
        let hit = self.check_obstacle_collision(state.x, state.y, state.z);
        (oob, hit)
    }
}

// src/reward.rs
use pyo3::prelude::*;
use crate::state::AircraftState;

/// Reward parameters (hardcoded defaults matching config, can be overridden).
pub struct RewardParams {
    pub pursuit_weight: f64,
    pub kill_reward: f64,
    pub collision_penalty: f64,
    pub alive_bonus: f64,
    pub kill_distance: f64,
    pub kill_angle: f64,
}

impl Default for RewardParams {
    fn default() -> Self {
        Self {
            pursuit_weight: 0.1,
            kill_reward: 10.0,
            collision_penalty: -10.0,
            alive_bonus: 0.01,
            kill_distance: 500.0,
            kill_angle: 0.524, // ~30 degrees
        }
    }
}

pub fn reward_inner(
    agent: &AircraftState,
    opponent: &AircraftState,
    params: &RewardParams,
    hit_obstacle: bool,
    out_of_bounds: bool,
) -> (f64, bool) {
    // Terminal: collision with obstacle or boundary
    if hit_obstacle || out_of_bounds {
        return (params.collision_penalty, true);
    }

    let distance = agent.distance_to(opponent);

    // Terminal: aircraft collision (within 10m)
    if distance < 10.0 {
        return (params.collision_penalty, true);
    }

    let mut reward = params.alive_bonus;

    // Pursuit reward: positive when nose-on and close
    let angle_to_opponent = agent.angle_to(opponent);
    let angle_factor = angle_to_opponent.cos(); // 1.0 nose-on, -1.0 away
    let dist_factor = 1.0 - (distance / 10000.0).min(1.0); // 1.0 close, 0.0 far
    reward += angle_factor * dist_factor * params.pursuit_weight;

    // Kill condition: nose-on within angle AND within distance
    if angle_to_opponent < params.kill_angle && distance < params.kill_distance {
        reward += params.kill_reward;
        return (reward, true);
    }

    (reward, false)
}

/// Compute reward for agent given opponent state.
/// Returns (reward, episode_done).
#[pyfunction]
#[pyo3(signature = (agent, opponent, hit_obstacle=false, out_of_bounds=false,
                    pursuit_weight=0.1, kill_reward=10.0, collision_penalty=-10.0,
                    alive_bonus=0.01, kill_distance=500.0, kill_angle=0.524))]
pub fn compute_reward(
    agent: &AircraftState,
    opponent: &AircraftState,

# python/aces/physics.py
"""Simplified 6-DOF flight physics for ACES MVP."""

from __future__ import annotations

import numpy as np


def _euler_to_rotation_matrix(roll: float, pitch: float, yaw: float) -> np.ndarray:
    """ZYX Euler angles to 3x3 rotation matrix (body-to-world)."""
    sr, cr = np.sin(roll), np.cos(roll)
    sp, cp = np.sin(pitch), np.cos(pitch)
    sy, cy = np.sin(yaw), np.cos(yaw)
    return np.array([
        [cy * cp, cy * sp * sr - sy * cr, cy * sp * cr + sy * sr],
        [sy * cp, sy * sp * sr + cy * cr, sy * sp * cr - cy * sr],
        [-sp,     cp * sr,                cp * cr],
    ])


class SimplifiedPhysics:
    """Simple flight dynamics: thrust along body X, drag, gravity."""

    GRAVITY = 9.81
    MASS = 5000.0         # kg
    MAX_THRUST = 50000.0  # N
    DRAG_COEFF = 0.02
    WING_AREA = 30.0      # m^2
    AIR_DENSITY = 1.225   # kg/m^3
    MAX_ANGULAR_RATE = 1.57  # rad/s (~90 deg/s)
    MIN_SPEED = 50.0      # m/s
    MAX_SPEED = 400.0     # m/s
    DT = 0.05             # seconds

    def __init__(self, **overrides):
        for key, val in overrides.items():
            attr = key.upper()
            if hasattr(self, attr):
                setattr(self, attr, val)

    def step(self, state: dict, action: np.ndarray) -> dict:
        """Advance one timestep.

        Args:
            state: dict with keys 'position', 'velocity', 'orientation'
                   (each np.ndarray of shape (3,), orientation = [roll, pitch, yaw])
            action: np.ndarray [throttle(0-1), pitch_cmd(-1,1), roll_cmd(-1,1), yaw_cmd(-1,1)]

        Returns:
            New state dict.
        """
        throttle = float(np.clip(action[0], 0.0, 1.0))
        pitch_cmd = float(np.clip(action[1], -1.0, 1.0))
        roll_cmd = float(np.clip(action[2], -1.0, 1.0))
        yaw_cmd = float(np.clip(action[3], -1.0, 1.0))

        pos = state["position"].copy()
        vel = state["velocity"].copy()
        ori = state["orientation"].copy()  # roll, pitch, yaw

        # Angular rates from commands
        p = roll_cmd * self.MAX_ANGULAR_RATE
        q = pitch_cmd * self.MAX_ANGULAR_RATE
        r = yaw_cmd * self.MAX_ANGULAR_RATE

        # Update orientation
        ori[0] += p * self.DT  # roll
        ori[1] = np.clip(ori[1] + q * self.DT, -np.pi / 2 + 0.01, np.pi / 2 - 0.01)
        ori[2] += r * self.DT  # yaw
        # Normalize yaw to [-pi, pi]
        ori[2] = (ori[2] + np.pi) % (2 * np.pi) - np.pi

        # Rotation matrix (body to world)
        rot = _euler_to_rotation_matrix(ori[0], ori[1], ori[2])

        # Thrust in body frame → world frame
        thrust_body = np.array([throttle * self.MAX_THRUST, 0.0, 0.0])
        thrust_world = rot @ thrust_body

        # Drag (opposes velocity)
        speed = np.linalg.norm(vel)
        if speed > 1.0:
            drag_mag = 0.5 * self.AIR_DENSITY * speed ** 2 * self.DRAG_COEFF * self.WING_AREA
            drag = -drag_mag * (vel / speed)
        else:
            drag = np.zeros(3)

        # Gravity (Z-up frame)
        gravity = np.array([0.0, 0.0, -self.GRAVITY * self.MASS])

        # Acceleration and integration
        accel = (thrust_world + drag + gravity) / self.MASS
        vel = vel + accel * self.DT

        # Speed clamping
        speed = np.linalg.norm(vel)
        if speed > self.MAX_SPEED:
            vel = vel * (self.MAX_SPEED / speed)
        elif speed < self.MIN_SPEED and speed > 0.1:
            vel = vel * (self.MIN_SPEED / speed)

        pos = pos + vel * self.DT

        return {
            "position": pos,
            "velocity": vel,
            "orientation": ori,
        }

# python/aces/env.py
"""Gymnasium environment for ACES 1v1 air combat."""

from __future__ import annotations

from pathlib import Path
from typing import Any

import gymnasium as gym
import numpy as np
import toml

from aces._core import AircraftState, Arena, Obstacle, compute_reward
from aces.physics import SimplifiedPhysics


# Observation layout:
#   own: pos(3) + vel(3) + ori(3) = 9
#   opponent relative: rel_pos(3) + rel_vel(3) + distance(1) = 7
#   obstacles: N * (rel_pos(3) + radius(1)) = N*4
# Total with 3 obstacles: 9 + 7 + 12 = 28

MAX_OBSTACLES = 8  # Pad to fixed size


class AcesEnv(gym.Env):
    metadata = {"render_modes": ["human"]}

    def __init__(
        self,
        config_path: str | None = None,
        max_episode_steps: int = 1000,
        render_mode: str | None = None,
    ):
        super().__init__()
        self.render_mode = render_mode

        # Load config
        if config_path is None:
            config_path = str(Path(__file__).parents[2] / "configs" / "default.toml")
        cfg = toml.load(config_path)

        # Arena
        ac = cfg["arena"]
        self.arena = Arena(
            x_max=ac["x_max"], y_max=ac["y_max"], z_max=ac["z_max"]
        )
        for obs_cfg in ac.get("obstacles", []):
            self.arena.add_obstacle(
                Obstacle(
                    x=obs_cfg["x"], y=obs_cfg["y"],
                    z=obs_cfg["z"], radius=obs_cfg["radius"],
                )
            )
        self.n_obstacles = self.arena.num_obstacles()

        # Physics
        pc = cfg["physics"]
        self.physics = SimplifiedPhysics(
            dt=pc["dt"],
            max_thrust=pc["max_thrust"],
            mass=pc["mass"],
            drag_coefficient=pc["drag_coefficient"],
            wing_area=pc["wing_area"],
            air_density=pc["air_density"],
            max_angular_rate=pc["max_angular_rate"],
            min_speed=pc["min_speed"],
            max_speed=pc["max_speed"],
        )
        self.dt = pc["dt"]

        # Reward params
        self.reward_cfg = cfg["reward"]

        # Training
        self.max_episode_steps = max_episode_steps

        # Spaces
        obs_dim = 9 + 7 + MAX_OBSTACLES * 4
        self.observation_space = gym.spaces.Box(
            low=-np.inf, high=np.inf, shape=(obs_dim,), dtype=np.float32
        )
        self.action_space = gym.spaces.Box(
            low=np.array([0.0, -1.0, -1.0, -1.0], dtype=np.float32),
            high=np.array([1.0, 1.0, 1.0, 1.0], dtype=np.float32),
        )

        # State
        self._agent_state: dict | None = None
        self._opponent_state: dict | None = None
        self._step_count = 0
        self._opponent_policy = None  # Set by trainer for self-play
        self._rng = np.random.default_rng()

    def set_opponent_policy(self, policy):
        """Set opponent policy for self-play. Must have .predict(obs) method."""
        self._opponent_policy = policy

    def _random_spawn(self) -> dict:
        """Generate random initial state within arena."""
        x = self._rng.uniform(1000.0, self.arena.x_max - 1000.0)
        y = self._rng.uniform(1000.0, self.arena.y_max - 1000.0)
        z = self._rng.uniform(500.0, self.arena.z_max - 500.0)
        yaw = self._rng.uniform(-np.pi, np.pi)
        speed = self._rng.uniform(100.0, 250.0)
        return {
            "position": np.array([x, y, z]),
            "velocity": np.array([
                speed * np.cos(yaw),
                speed * np.sin(yaw),
                0.0,
            ]),
            "orientation": np.array([0.0, 0.0, yaw]),
        }

    def _state_to_aircraft(self, state: dict) -> AircraftState:
        p, v, o = state["position"], state["velocity"], state["orientation"]
        return AircraftState(
            x=p[0], y=p[1], z=p[2],
            vx=v[0], vy=v[1], vz=v[2],
            roll=o[0], pitch=o[1], yaw=o[2],
        )

    def _get_obs(self, agent_state: dict, opponent_state: dict) -> np.ndarray:
        """Build observation vector for the agent."""
        ap = agent_state["position"]
        av = agent_state["velocity"]
        ao = agent_state["orientation"]

        op = opponent_state["position"]
        ov = opponent_state["velocity"]

        rel_pos = op - ap
        rel_vel = ov - av
        distance = np.linalg.norm(rel_pos)

        obs = np.zeros(9 + 7 + MAX_OBSTACLES * 4, dtype=np.float32)

        # Normalize positions by arena scale
        scale = np.array([self.arena.x_max, self.arena.y_max, self.arena.z_max])

        obs[0:3] = ap / scale
        obs[3:6] = av / 400.0  # normalize by max speed
        obs[6:9] = ao / np.pi  # normalize to [-1, 1]
        obs[9:12] = rel_pos / scale
        obs[12:15] = rel_vel / 400.0
        obs[15] = distance / np.linalg.norm(scale)

        # Obstacles (relative to agent)
        obstacles = self.arena.get_obstacles()
        for i, obs_obj in enumerate(obstacles):
            if i >= MAX_OBSTACLES:
                break
            base = 16 + i * 4
            obs_pos = np.array(obs_obj.position())
            obs[base:base + 3] = (obs_pos - ap) / scale
            obs[base + 3] = obs_obj.radius / 1000.0

        return obs

    def _get_opponent_obs(self) -> np.ndarray:
        """Observation from opponent's perspective."""
        return self._get_obs(self._opponent_state, self._agent_state)

    def reset(
        self, *, seed: int | None = None, options: dict | None = None
    ) -> tuple[np.ndarray, dict]:
        super().reset(seed=seed)
        if seed is not None:
            self._rng = np.random.default_rng(seed)

        self._agent_state = self._random_spawn()
        self._opponent_state = self._random_spawn()
        self._step_count = 0

        obs = self._get_obs(self._agent_state, self._opponent_state)
        return obs, {}

    def step(
        self, action: np.ndarray
    ) -> tuple[np.ndarray, float, bool, bool, dict[str, Any]]:
        self._step_count += 1

        # Get opponent action
        if self._opponent_policy is not None:
            opp_obs = self._get_opponent_obs()
            opp_action, _ = self._opponent_policy.predict(opp_obs, deterministic=False)
        else:
            opp_action = self.action_space.sample()

        # Step physics for both aircraft
        self._agent_state = self.physics.step(self._agent_state, action)
        self._opponent_state = self.physics.step(self._opponent_state, opp_action)

        # Check collisions
        agent_ac = self._state_to_aircraft(self._agent_state)
        opp_ac = self._state_to_aircraft(self._opponent_state)

        agent_oob, agent_obs_hit = self.arena.check_state(agent_ac)
        opp_oob, opp_obs_hit = self.arena.check_state(opp_ac)

        # Compute reward for agent
        reward, agent_done = compute_reward(
            agent_ac, opp_ac,
            hit_obstacle=agent_obs_hit,
            out_of_bounds=agent_oob,
            **self.reward_cfg,
        )

        # Opponent terminal → agent gets bonus, episode ends
        opp_dead = opp_oob or opp_obs_hit
        if opp_dead and not agent_done:
            reward += 5.0
            agent_done = True

        truncated = self._step_count >= self.max_episode_steps
        terminated = agent_done

        obs = self._get_obs(self._agent_state, self._opponent_state)

        info = {
            "agent_pos": self._agent_state["position"].copy(),
            "opponent_pos": self._opponent_state["position"].copy(),
            "agent_speed": float(np.linalg.norm(self._agent_state["velocity"])),
            "step": self._step_count,
        }

        return obs, float(reward), terminated, truncated, info

# python/aces/trainer.py
"""Self-play PPO training for ACES."""

from __future__ import annotations

import copy
from pathlib import Path
from typing import Any, Callable

import numpy as np
from stable_baselines3 import PPO
from stable_baselines3.common.callbacks import BaseCallback

from aces.env import AcesEnv


class OpponentUpdateCallback(BaseCallback):
    """Periodically copies the current policy weights to the opponent."""

    def __init__(self, update_interval: int = 10000, verbose: int = 0):
        super().__init__(verbose)
        self.update_interval = update_interval
        self.update_count = 0

    def _on_step(self) -> bool:
        if self.num_timesteps % self.update_interval < self.model.n_steps:
            # Copy current policy to opponent in env
            policy_state = copy.deepcopy(self.model.policy.state_dict())
            envs = self.training_env.envs
            for env in envs:
                unwrapped = env.unwrapped
                if hasattr(unwrapped, "_set_opponent_weights"):
                    unwrapped._set_opponent_weights(policy_state)
            self.update_count += 1
            if self.verbose:
                print(f"[SelfPlay] Opponent updated (#{self.update_count})")
        return True


class StateCallback(BaseCallback):
    """Collects per-step state for visualization."""

    def __init__(self, state_callback: Callable | None = None, verbose: int = 0):
        super().__init__(verbose)
        self.state_callback = state_callback

    def _on_step(self) -> bool:
        if self.state_callback is not None:
            infos = self.locals.get("infos", [])
            for info in infos:
                if "agent_pos" in info:
                    self.state_callback(info)
        return True


class SelfPlayTrainer:
    """Trains a PPO agent via self-play in the ACES environment."""

    def __init__(
        self,
        config_path: str | None = None,
        total_timesteps: int = 500_000,
        n_steps: int = 2048,
        batch_size: int = 64,
        learning_rate: float = 3e-4,
        gamma: float = 0.99,
        gae_lambda: float = 0.95,
        clip_range: float = 0.2,
        n_epochs: int = 10,
        opponent_update_interval: int = 10000,
        max_episode_steps: int = 1000,
        state_callback: Callable | None = None,
    ):
        self.total_timesteps = total_timesteps
        self.opponent_update_interval = opponent_update_interval
        self.state_callback = state_callback
        self.opponent_update_count = 0

        self.env = AcesEnv(
            config_path=config_path,
            max_episode_steps=max_episode_steps,
        )

        self.model = PPO(
            "MlpPolicy",
            self.env,
            learning_rate=learning_rate,
            n_steps=n_steps,
            batch_size=batch_size,
            gamma=gamma,
            gae_lambda=gae_lambda,
            clip_range=clip_range,
            n_epochs=n_epochs,
            verbose=0,
        )

        # Give env a way to use the opponent policy
        self._setup_opponent()

    def _setup_opponent(self):
        """Wire the opponent policy into the environment."""
        # Create a separate PPO for opponent predictions
        self._opponent_model = PPO(
            "MlpPolicy", self.env, verbose=0,
        )
        # Give env a method to receive weight updates
        env = self.env
        opponent_model = self._opponent_model

        def set_opponent_weights(state_dict):
            opponent_model.policy.load_state_dict(state_dict)

        env._set_opponent_weights = set_opponent_weights
        env.set_opponent_policy(opponent_model.policy)

    def train(self) -> PPO:
        opp_cb = OpponentUpdateCallback(
            update_interval=self.opponent_update_interval, verbose=1
        )
        state_cb = StateCallback(state_callback=self.state_callback)

        self.model.learn(
            total_timesteps=self.total_timesteps,
            callback=[opp_cb, state_cb],
        )
        self.opponent_update_count = opp_cb.update_count
        return self.model

    def save(self, path: str = "aces_model"):

# python/aces/vis.py
"""Real-time 3D visualization for ACES using PyVista."""

from __future__ import annotations

from collections import deque

import numpy as np
import pyvista as pv


def _make_aircraft_mesh(scale: float = 100.0) -> pv.PolyData:
    """Create a simple cone+arrow aircraft mesh."""
    cone = pv.Cone(
        center=(0, 0, 0), direction=(1, 0, 0),
        height=scale, radius=scale * 0.3, resolution=16,
    )
    return cone


class AcesVisualizer:
    """Real-time 3D viewer for the ACES simulation."""

    def __init__(
        self,
        arena_bounds: tuple[float, float, float],
        obstacles: list[tuple[float, float, float, float]],
        trail_length: int = 200,
        window_size: tuple[int, int] = (1280, 960),
        off_screen: bool = False,
    ):
        self._arena_bounds = arena_bounds
        self._obstacles_data = obstacles
        self._trail_length = trail_length
        self._off_screen = off_screen

        self._agent_trail: deque[np.ndarray] = deque(maxlen=trail_length)
        self._opponent_trail: deque[np.ndarray] = deque(maxlen=trail_length)

        # Create plotter
        self._plotter = pv.Plotter(
            window_size=window_size,
            off_screen=off_screen,
        )
        self._setup_scene()
        self._agent_actor = None
        self._opponent_actor = None
        self._agent_trail_actor = None
        self._opponent_trail_actor = None

    def _setup_scene(self):
        """Initialize the static scene elements."""
        pl = self._plotter
        pl.set_background("black")

        # Arena bounding box
        x, y, z = self._arena_bounds
        box = pv.Box(bounds=(0, x, 0, y, 0, z))
        pl.add_mesh(box, style="wireframe", color="gray", opacity=0.3, line_width=1)

        # Ground plane
        ground = pv.Plane(
            center=(x / 2, y / 2, 0), direction=(0, 0, 1),
            i_size=x, j_size=y,
        )
        pl.add_mesh(ground, color="darkgreen", opacity=0.2)

        # Obstacles
        for ox, oy, oz, r in self._obstacles_data:
            sphere = pv.Sphere(radius=r, center=(ox, oy, oz))
            pl.add_mesh(sphere, color="red", opacity=0.4)

        # Camera setup
        pl.camera_position = [
            (x / 2, -y * 0.8, z * 1.5),  # position
            (x / 2, y / 2, z / 3),        # focal point
            (0, 0, 1),                     # view up
        ]

    def push_state(self, state: dict):
        """Push a simulation state for rendering."""
        if "agent_pos" in state:
            self._agent_trail.append(state["agent_pos"].copy())
        if "opponent_pos" in state:
            self._opponent_trail.append(state["opponent_pos"].copy())

    def render_frame(self):
        """Render one frame with current state."""
        pl = self._plotter

        # Remove old aircraft actors
        if self._agent_actor is not None:
            pl.remove_actor(self._agent_actor)
        if self._opponent_actor is not None:
            pl.remove_actor(self._opponent_actor)
        if self._agent_trail_actor is not None:
            pl.remove_actor(self._agent_trail_actor)
        if self._opponent_trail_actor is not None:
            pl.remove_actor(self._opponent_trail_actor)

        # Draw aircraft
        if self._agent_trail:
            pos = self._agent_trail[-1]
            mesh = _make_aircraft_mesh(scale=150.0)
            mesh.translate(pos, inplace=True)
            self._agent_actor = pl.add_mesh(
                mesh, color="cyan", name="_agent"
            )

        if self._opponent_trail:
            pos = self._opponent_trail[-1]
            mesh = _make_aircraft_mesh(scale=150.0)
            mesh.translate(pos, inplace=True)
            self._opponent_actor = pl.add_mesh(
                mesh, color="orange", name="_opponent"
            )

        # Draw trails
        if len(self._agent_trail) >= 2:
            points = np.array(list(self._agent_trail))
            line = pv.Spline(points, n_points=len(points))
            self._agent_trail_actor = pl.add_mesh(
                line, color="cyan", opacity=0.6, line_width=2, name="_agent_trail"
            )

        if len(self._opponent_trail) >= 2:
            points = np.array(list(self._opponent_trail))
            line = pv.Spline(points, n_points=len(points))
            self._opponent_trail_actor = pl.add_mesh(
                line, color="orange", opacity=0.6, line_width=2, name="_opponent_trail"
            )

        if not self._off_screen:
            pl.update()

    def show(self):
        """Open the interactive window."""
        self._plotter.show(interactive_update=True, auto_close=False)

    def close(self):
        self._plotter.close()
```

# scripts/run.py
"""ACES — Train and visualize in real time."""

from __future__ import annotations

import argparse
import threading
import time
from pathlib import Path

import numpy as np
import toml

from aces.env import AcesEnv
from aces.trainer import SelfPlayTrainer
from aces.vis import AcesVisualizer


def main():
    parser = argparse.ArgumentParser(description="ACES: Air Combat Engagement Simulation")
    parser.add_argument("--config", default=None, help="Path to config TOML file")
    parser.add_argument("--timesteps", type=int, default=None, help="Override total_timesteps")
    parser.add_argument("--no-vis", action="store_true", help="Disable visualization")
    parser.add_argument("--save-path", default="aces_model", help="Model save path")
    args = parser.parse_args()

    # Load config
    config_path = args.config
    if config_path is None:
        config_path = str(Path(__file__).parent.parent / "configs" / "default.toml")
    cfg = toml.load(config_path)

    # Extract visualization params
    vis_cfg = cfg.get("visualization", {})
    fps = vis_cfg.get("fps", 30)
    trail_length = vis_cfg.get("trail_length", 200)
    window_size = tuple(vis_cfg.get("window_size", [1280, 960]))

    # Training params
    tc = cfg.get("training", {})
    total_timesteps = args.timesteps or tc.get("total_timesteps", 500_000)

    # Setup visualizer
    vis = None
    if not args.no_vis:
        ac = cfg["arena"]
        obstacles = [
            (o["x"], o["y"], o["z"], o["radius"])
            for o in ac.get("obstacles", [])
        ]
        vis = AcesVisualizer(
            arena_bounds=(ac["x_max"], ac["y_max"], ac["z_max"]),
            obstacles=obstacles,
            trail_length=trail_length,
            window_size=window_size,
        )

    # State callback: push sim states to visualizer
    def on_state(info: dict):
        if vis is not None:
            vis.push_state(info)

    # Create trainer
    trainer = SelfPlayTrainer(
        config_path=config_path,
        total_timesteps=total_timesteps,
        n_steps=tc.get("n_steps", 2048),
        batch_size=tc.get("batch_size", 64),
        learning_rate=tc.get("learning_rate", 3e-4),
        gamma=tc.get("gamma", 0.99),
        gae_lambda=tc.get("gae_lambda", 0.95),
        clip_range=tc.get("clip_range", 0.2),
        n_epochs=tc.get("n_epochs", 10),
        opponent_update_interval=tc.get("opponent_update_interval", 10000),
        max_episode_steps=tc.get("max_episode_steps", 1000),
        state_callback=on_state,
    )

    if vis is not None:
        # Run training in background thread, vis in main thread
        vis.show()
        train_done = threading.Event()

        def train_thread():
            print(f"[ACES] Training for {total_timesteps} timesteps...")
            trainer.train()
            trainer.save(args.save_path)
            print(f"[ACES] Training complete. Model saved to {args.save_path}")
            train_done.set()

        t = threading.Thread(target=train_thread, daemon=True)
        t.start()

        frame_interval = 1.0 / fps
        while not train_done.is_set():
            vis.render_frame()
            time.sleep(frame_interval)

        vis.render_frame()  # Final frame
        print("[ACES] Close the visualization window to exit.")
        vis._plotter.show()  # Block until window closed
    else:
        # No visualization — just train
        print(f"[ACES] Training for {total_timesteps} timesteps (no visualization)...")
        trainer.train()
        trainer.save(args.save_path)
        print(f"[ACES] Training complete. Model saved to {args.save_path}")


if __name__ == "__main__":
    main()

```python
# python/aces/jsbsim_physics.py
"""JSBSim-backed flight physics for ACES.

This module wraps JSBSim to provide the same interface as SimplifiedPhysics,
so it can be swapped in via config. Requires `pip install jsbsim`.
"""

from __future__ import annotations

import numpy as np

try:
    import jsbsim
except ImportError:
    jsbsim = None


class JSBSimPhysics:
    """Flight dynamics using the JSBSim FDM.

    Uses the default F-16 (or c172p) aircraft model from JSBSim.
    Exposes the same step(state, action) interface as SimplifiedPhysics.
    """

    DT = 0.05  # seconds per step

    def __init__(self, aircraft: str = "f16", dt: float = 0.05):
        if jsbsim is None:
            raise ImportError("jsbsim is required: pip install jsbsim")

        self.DT = dt
        self._dt_jsbsim = dt
        self._aircraft = aircraft

        # Initialize JSBSim FDM
        self._fdm = jsbsim.FGFDMExec(None)
        self._fdm.set_debug_level(0)
        self._fdm.load_model(aircraft)
        self._fdm.set_dt(dt)

        # Disable JSBSim output
        self._fdm.disable_output()

    def initial_state(
        self,
        position: np.ndarray,
        speed: float = 200.0,
        heading: float = 0.0,
    ) -> dict:
        """Set initial conditions and return state dict."""
        fdm = self._fdm

        # Convert from our sim coords (meters, Z-up) to JSBSim (ft, NED)
        lat, lon = 0.0, 0.0  # Arbitrary lat/lon
        alt_ft = position[2] * 3.28084

        fdm["ic/h-sl-ft"] = alt_ft
        fdm["ic/vt-kts"] = speed * 1.94384  # m/s to knots
        fdm["ic/psi-true-deg"] = np.degrees(heading)
        fdm["ic/phi-deg"] = 0.0
        fdm["ic/theta-deg"] = 0.0

        fdm.run_ic()

        return self._extract_state(position[:2])

    def step(self, state: dict, action: np.ndarray) -> dict:
        """Advance one step. action = [throttle, pitch, roll, yaw]."""
        fdm = self._fdm

        throttle = float(np.clip(action[0], 0.0, 1.0))
        elevator = float(np.clip(action[1], -1.0, 1.0))
        aileron = float(np.clip(action[2], -1.0, 1.0))
        rudder = float(np.clip(action[3], -1.0, 1.0))

        fdm["fcs/throttle-cmd-norm"] = throttle
        fdm["fcs/elevator-cmd-norm"] = elevator
        fdm["fcs/aileron-cmd-norm"] = aileron
        fdm["fcs/rudder-cmd-norm"] = rudder

        fdm.run()

        return self._extract_state(state["position"][:2])

    def _extract_state(self, ref_xy: np.ndarray) -> dict:
        """Extract state dict from JSBSim FDM."""
        fdm = self._fdm

        # Position: use propagated position (convert ft to m)
        alt_m = fdm["position/h-sl-ft"] / 3.28084
        # For X/Y we use velocity integration (JSBSim uses lat/lon internally)
        vn = fdm["velocities/v-north-fps"] / 3.28084  # m/s
        ve = fdm["velocities/v-east-fps"] / 3.28084
        vd = fdm["velocities/v-down-fps"] / 3.28084

        pos = np.array([
            ref_xy[0] + ve * self.DT,
            ref_xy[1] + vn * self.DT,
            alt_m,
        ])

        vel = np.array([ve, vn, -vd])  # Convert NED to Z-up

        roll = np.radians(fdm["attitude/phi-deg"])
        pitch = np.radians(fdm["attitude/theta-deg"])
        yaw = np.radians(fdm["attitude/psi-deg"])

        return {
            "position": pos,
            "velocity": vel,
            "orientation": np.array([roll, pitch, yaw]),
        }
```
